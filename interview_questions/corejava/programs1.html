//For example, 371 is an Armstrong number since 3**3 + 7**3 + 1**3 = 371.
public class Armstorng 
{
	public static void main(String[] args) 
	{
		int n;
		int temp;

Write a Java Program to whether number is palindrome or not (Input-12521, Output: Yes) 

package com.core;
// A String/Number is a palindrome if it remains unchanged when it reversed
public class Palindrome 
{
	public static void main(String[] args) 
	{
		int num = 123212;
		String str = "DAD";
		palindrome Number(num);
		palindrome String(str);

	}
	private static void palindromeNumber(int num) 
	{
		int n = num;
		int revNumber = 0;
		while (num != 0) 
		{
			revNumber = (revNumber * 10) + num % 10;
			num = num / 10;
		}
		if (revNumber == n)
			System.out.println("Given Number " + n + " is a palindrome");
		else
			System.out.println("Given Number " + n + " is NOT a palindrome");
	}
	private static void palindromeString(String str) 
	{
		String str1 = str;
		String revStr = "";
		int len = str.length();
		char ch;
		char[] charStr = str.toCharArray();

		for (int i = 0; i < len / 2; i++) 
		{
			ch = charStr[i];
			charStr[i] = charStr[len - 1 - i];
			charStr[len - 1 - i] = ch;
		}
		revStr = String.valueOf(charStr);

		if (str1.equals(revStr))
			System.out.println("Given String " + str + " is a palindrome");
		else
			System.out.println("Given String " + str + " is NOT a palindrome");
	}

}

Write a Java Program for a Factorial of a number 

package com.core;
//Factorial number is multiplication of all positive integer from one to that number.
public class Factorial 
{
	int fact = 1;
	public static void main(String[] args) 
	{
		int num = 5;
		int fact = 1;
		for (int i = 1; i <= num; i++)
		fact = fact * i;
		System.out.println("Factorial of a number is :" + fact);
		System.out.println("Factorial using recursion : " + factRecursion(num));
	}
	private static int factRecursion(int num) 
	{
		if (num == 1)
			return 1;
		else
			return num * factRecursion(num - 1);
	}
}

/**
Static members are not inherited in java as they are the properties of class and they are loaded in class area,
they had nothing to do with object creation. However only the child classes can access the static members of their parent classes.
*/

package com.core;
public class StaticSuper {
	public static int i = 10;
	public static String name = "BrightITCareer";
	public static void display() {
		System.out.println("i  : +" + i);
	}
}
package com.core;
public class StaticSub extends StaticSuper{

	public static void main(String[] args){
		System.out.println("In Sub Class: "+i);
		System.out.println("In Sub Class: "+name);
		display();
	}
}

// Sorting in order of price using Comparator and anonymous inner class
private static void comparatorSortIntegers() {
	List plist = new ArrayList();
	plist.add(new Price(10, "Audi"));
	plist.add(new Price(20, "BMW"));
	plist.add(new Price(36, "Hyundai"));
	plist.add(new Price(23, "Audi"));
	plist.add(new Price(27, "Benz"));
	Collections.sort(plist, new Comparator() {
		@Override
		public int compare(Price p1, Price p2) {
			if (p1 == null || p2 == null) {
			throw new NullPointerException("Objects are null");
			} else if (!p1.getClass().equals(p2.getClass())) {
				throw new ClassCastException("Objects are not of the same Type, Not mutually comparable");
			} else if (p1.getPrice() > p2.getPrice()) {
				return 1;
			} else if (p1.getPrice() < p2.getPrice()) {
				return -1;
			} else
				return 0;
		}
	});
	System.out.println("Sort in order of Price -Comparator and anonymous Inner class ");
	for (Price i : plist) {
			System.out.println(i.getPrice());
	}
	System.out.println("\n");
}

		}
	});

		System.out.println(p.getBrand());
