//  use type to locate all the checkboxes on the page
	@Test
	public void testCheckBoxByLabel() throws InterruptedException {
		driver.get(url);
		String xPath = "//input[@type='checkbox']";
		List checkBoxGroup = driver.findElements(By.xpath(xPath));
		for (int i = 0; i < checkBoxGroup.size(); i++) {
			if (checkBoxGroup.get(i).isEnabled())
				System.out.println("The CheckBox : " + checkBoxGroup.get(i).getAttribute("value") + "   isenabled");
		}
		System.out.println("CheckBox Count : " + checkBoxGroup.size());
	}

List down all the values of DropDown/ListBox and find the selected value 

// Lists down all the option values and finds the selected value
	@Test
	public void testSelectOptionDropDown() {
		driver.get(url);
		WebElement dropDownEle = driver.findElement(By.id("ddlcountry"));
		Select selectDropDown = new Select(dropDownEle);
		List listOptions = selectDropDown.getOptions();
		for (int i = 0; i < listOptions.size(); i++) {

			String str = listOptions.get(i).getText();
			System.out.println("getText() is : " + str);
			if (listOptions.get(i).isSelected())
				System.out.println("Selected value is : " + str);
		}
	}
